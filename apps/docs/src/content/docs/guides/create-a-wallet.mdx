## Install the packages

WalletDK ships as a transport package, one per platform, plus
`@lightninglabs/walletdk-react`, the React bindings. All are ES modules with
full TypeScript declarations.

<PlatformTabs>
<div data-label="Web">

The web transport, `@lightninglabs/walletdk-web`, bundles a WASM module for
cryptography and uses OPFS for storage. There are no native dependencies and
no server-side component, so a modern browser is all you need at runtime.

```bash title="Terminal"
# Install the browser runtime and the React bindings.
npm install @lightninglabs/walletdk-web @lightninglabs/walletdk-react
```

</div>
<div data-label="React Native">

The React Native transport, `@lightninglabs/walletdk-react-native`, wraps a
wallet runtime that is compiled into your app binary rather than fetched at
runtime.

```bash title="Terminal"
# Install the native transport and the React bindings.
npm install @lightninglabs/walletdk-react-native @lightninglabs/walletdk-react
```

See [Installation](/react-native/get-started/installation/) for staging the
native runtime binaries before your first build.

</div>
</PlatformTabs>

## Add the provider

Wrap your app in `WalletDKProvider` so every component below it can reach the
wallet through hooks.

Call `start(defaultConfig('signet'))` to boot the daemon after the runtime
loads. Switch to `defaultConfig('mainnet')` only once you have key backup and
recovery in place.

<PlatformTabs>
<div data-label="Web">

`createWebClient()` returns a browser client. Build it outside the provider
and pass it via `client`:

```tsx title="App.tsx"
import { useEffect } from 'react';
import { createWebClient, defaultConfig } from '@lightninglabs/walletdk-web';
import { WalletDKProvider, useWalletRuntime } from '@lightninglabs/walletdk-react';

const client = createWebClient({ runtimeBaseUrl: '/walletdk/' });

function WalletApp({ children }: { children: React.ReactNode }) {
  const { start, phase } = useWalletRuntime();

  useEffect(() => {
    if (phase === 'runtimeReady') {
      start(defaultConfig('signet'));
    }
  }, [phase, start]);

  return <>{children}</>;
}

export function App() {
  return (
    <WalletDKProvider client={client}>
      <WalletApp>
        {/* your app */}
      </WalletApp>
    </WalletDKProvider>
  );
}
```

</div>
<div data-label="React Native">

`createNativeClient` takes no options, so pass it directly to the provider's
`createClient` factory prop instead of building a client beforehand:

```tsx title="App.tsx"
import { useEffect } from 'react';
import { createNativeClient, defaultConfig } from '@lightninglabs/walletdk-react-native';
import { WalletDKProvider, useWalletRuntime } from '@lightninglabs/walletdk-react';

function WalletApp({ children }: { children: React.ReactNode }) {
  const { start, phase } = useWalletRuntime();

  useEffect(() => {
    if (phase === 'runtimeReady') {
      start(defaultConfig('signet'));
    }
  }, [phase, start]);

  return <>{children}</>;
}

export function App() {
  return (
    <WalletDKProvider createClient={createNativeClient}>
      <WalletApp>
        {/* your app */}
      </WalletApp>
    </WalletDKProvider>
  );
}
```

## Create the wallet

Call `useWalletBootstrap()` inside the provider to create or unlock a wallet.
When `phase` is `'needsWallet'`, no wallet exists yet.

`createWallet({ password })` generates a fresh HD seed, encrypts it with the
password, and stores it locally. The password is whatever you pass in; the
SDK does not enforce a minimum length or complexity, so validate it yourself
before calling `createWallet`. The whole operation runs on-device and the
seed never leaves it. After it resolves, `phase` becomes `'ready'`.

The resolved result includes `mnemonic`, the wallet's recovery phrase. This is
the only time the SDK ever returns it, so show it to the user immediately and
prompt them to write it down before continuing. To rebuild a wallet from an
existing recovery phrase instead of generating a new one, see
[Restore a wallet](/guides/restore-a-wallet/).

```tsx title="Onboard.tsx"
import { useState } from 'react';
import { useWalletBootstrap, useWalletDK } from '@lightninglabs/walletdk-react';

function Onboard() {
  const { createWallet, operations } = useWalletBootstrap();
  const { phase } = useWalletDK();
  const [mnemonic, setMnemonic] = useState<string[] | null>(null);

  const handleCreate = async () => {
    // Generate the HD seed on-device, encrypted with the password.
    const result = await createWallet({ password: 'a-strong-password' });
    // Show the mnemonic once, right after creation: it is not retrievable later.
    setMnemonic(result.mnemonic);
  };

  if (phase !== 'needsWallet') {
    return null;
  }

  return (
    <>
      <button onClick={handleCreate} disabled={operations.createWallet.busy}>
        Create wallet
      </button>
      {operations.createWallet.error && (
        <p role="alert">{operations.createWallet.error}</p>
      )}
      {mnemonic && (
        <p>Write down your recovery phrase: {mnemonic.join(' ')}</p>
      )}
    </>
  );
}
```
